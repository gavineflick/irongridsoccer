<script>
const BG  = ["#EEEDFE","#E1F5EE","#FAECE7","#FBEAF0","#E6F1FB","#EAF3DE","#FAEEDA","#FCEBEB","#F1EFE8","#E1F5EE","#EEEDFE","#FAECE7","#E6F1FB","#FBEAF0","#EAF3DE","#FAEEDA","#FCEBEB","#EEEDFE","#E1F5EE","#FAECE7","#FBEAF0","#E6F1FB","#EAF3DE","#FAEEDA","#FCEBEB"];
const FG  = ["#3C3489","#085041","#712B13","#72243E","#0C447C","#27500A","#633806","#791F1F","#444441","#085041","#3C3489","#712B13","#0C447C","#72243E","#27500A","#633806","#791F1F","#3C3489","#085041","#712B13","#72243E","#0C447C","#27500A","#633806","#791F1F"];
const DOT = ["#534AB7","#0F6E56","#993C1D","#993556","#185FA5","#3B6D11","#854F0B","#A32D2D","#5F5E5A","#0F6E56","#534AB7","#993C1D","#185FA5","#993556","#3B6D11","#854F0B","#A32D2D","#534AB7","#0F6E56","#993C1D","#993556","#185FA5","#3B6D11","#854F0B","#A32D2D"];
let names = ["Alex","Blake","Casey","Dana","Elliot","Frankie","Harper","Jordan","Morgan","Riley"];
let payouts = [{label:"Halftime",pct:25},{label:"Final score",pct:75}];
let lastNames=null,lastOwner=null,lastHome=null,lastAway=null;
function shuffle(a){const b=[...a];for(let i=b.length-1;i>0;i--){const j=Math.floor(Math.random()*(i+1));[b[i],b[j]]=[b[j],b[i]];}return b;}
function renderNames(){
  document.getElementById('name-grid').innerHTML=names.map((n,i)=>`
    <div class="name-row"><span class="name-num">${i+1}</span>
    <input type="text" value="${n.replace(/"/g,'&quot;')}" placeholder="Name ${i+1}" oninput="names[${i}]=this.value;updatePool()">
    <button class="remove-btn" onclick="removeName(${i})" aria-label="Remove">&times;</button></div>`).join('');
}
function addName(){
  if(names.length>=25){document.getElementById('name-err').textContent='Maximum 25 participants.';return;}
  document.getElementById('name-err').textContent='';
  names.push('');renderNames();updatePool();
  document.querySelectorAll('.name-row input')[names.length-1].focus();
}
function removeName(i){if(names.length<=1)return;names.splice(i,1);renderNames();updatePool();}
function updatePool(){
  const b=parseFloat(document.getElementById('buyin').value)||0;
  const n=names.filter(x=>x.trim()).length;
  document.getElementById('total-pool').textContent='$'+(n*b).toLocaleString();
  updatePayoutStatus();
}
function renderPayouts(){
  document.getElementById('payout-rows').innerHTML=payouts.map((p,i)=>`
    <div class="payout-entry">
      <div>${i===0?'<label class="field-label">Checkpoint label</label>':''}
        <input type="text" value="${p.label.replace(/"/g,'&quot;')}" placeholder="e.g. Halftime" oninput="payouts[${i}].label=this.value"></div>
      <div>${i===0?'<label class="field-label">% of pool</label>':''}
        <input type="number" value="${p.pct}" min="0" max="100" step="1" oninput="payouts[${i}].pct=parseFloat(this.value)||0;updatePayoutStatus()"></div>
      <button class="remove-btn" style="padding-top:${i===0?'20px':'0'}" onclick="removePayout(${i})" aria-label="Remove">&times;</button>
    </div>`).join('');
  updatePayoutStatus();
}
function addPayout(){payouts.push({label:'',pct:0});renderPayouts();}
function removePayout(i){if(payouts.length<=1)return;payouts.splice(i,1);renderPayouts();}
function updatePayoutStatus(){
  const sum=payouts.reduce((a,p)=>a+(parseFloat(p.pct)||0),0);
  document.getElementById('payout-status').innerHTML=Math.round(sum)===100
    ?'<span class="ok">✓ Payouts total 100%</span>'
    :`<span class="err">Payouts total ${sum.toFixed(0)}% — must equal 100%</span>`;
}
function buildGrid(vn){
  const n=vn.length,total=25,idx=shuffle(Array.from({length:total},(_,i)=>i));
  const owner=new Array(total).fill(null);
  const base=Math.floor(total/n),extra=total%n;
  let pos=0;
  vn.forEach((_,pi)=>{const c=base+(pi<extra?1:0);for(let s=0;s<c;s++)owner[idx[pos++]]=pi;});
  return{owner,home:shuffle([0,1,2,3,4]),away:shuffle([0,1,2,3,4])};
}
function renderResult(vn,owner,home,away){
  const buyin=parseFloat(document.getElementById('buyin').value)||0;
  const total=vn.length*buyin;
  const base=Math.floor(25/vn.length),extra=25%vn.length;
  let h='';
  h+=`<div class="section-label">Summary</div><div class="metrics">
    <div class="metric"><div class="metric-label">Total pool</div><div class="metric-value">$${total.toLocaleString()}</div></div>
    <div class="metric"><div class="metric-label">Participants</div><div class="metric-value">${vn.length}</div></div>
    <div class="metric"><div class="metric-label">Buy-in</div><div class="metric-value">$${buyin.toLocaleString()}</div></div>
    <div class="metric"><div class="metric-label">Squares each</div><div class="metric-value">${base}${extra>0?'–'+(base+1):''}</div></div></div>`;
  h+=`<div class="section-label">Payouts</div><div class="payouts-row">`;
  payouts.forEach((p,i)=>{
    const amt=Math.round(total*(parseFloat(p.pct)||0)/100);
    const cls=i===0?'ht':(i===payouts.length-1?'fn':'');
    h+=`<div class="payout-chip ${cls}"><div class="chip-label">${p.label||'Checkpoint '+(i+1)}</div><div class="chip-amount">$${amt.toLocaleString()}</div><div class="chip-sub">${p.pct}% of pool</div></div>`;
  });
  h+='</div>';
  h+=`<div class="section-label">Assignments</div><div class="pills">`;
  vn.forEach((name,i)=>{
    const sqs=owner.reduce((a,o,idx)=>{if(o===i)a.push(idx);return a;},[]);
    const coords=sqs.map(idx=>`${home[idx%5]}-${away[Math.floor(idx/5)]}`).join(' ');
    h+=`<div class="pill"><div class="pdot" style="background:${DOT[i]}"></div><span class="pill-name">${name}</span><span class="pill-squares">${coords}</span></div>`;
  });
  h+='</div>';
  h+=`<div class="action-btns">
    <button class="reroll-btn" onclick="reroll()">↺ Re-randomize</button>
    <button class="print-btn" onclick="window.print()">🖨 Print</button></div>`;
  h+=`<div class="section-label">Grid — home team (columns) | away team (rows)</div>
  <div class="grid-outer"><div class="axis-header"><div style="width:37px;flex-shrink:0"></div>
  <div class="axis-title-h">← Home team score (last digit) →</div></div>
  <table class="sq"><thead><tr><th class="corner ax"></th>`;
  home.forEach(d=>{h+=`<th class="ax">${d}</th>`;});
  h+='</tr></thead><tbody>';
  for(let r=0;r<5;r++){
    h+=`<tr><th class="ax">${away[r]}</th>`;
    for(let c=0;c<5;c++){
      const idx=r*5+c,o=owner[idx];
      if(o!==null){
        h+=`<td class="sq-cell" style="background:${BG[o]};color:${FG[o]}"><div class="sq-name">${vn[o]}</div><div class="sq-coord">${home[c]}-${away[r]}</div></td>`;
      }else{
        h+=`<td class="sq-cell" style="background:#f2f2f0;color:#aaa"><div class="sq-name" style="font-size:9px">House</div><div class="sq-coord">${home[c]}-${away[r]}</div></td>`;
      }
    }
    h+='</tr>';
  }
  h+='</tbody></table></div>';
  h+=`<div class="section-label">Legend</div><div class="legend">`;
  vn.forEach((name,i)=>{h+=`<div class="legend-item"><div class="ldot" style="background:${BG[i]};border-color:${DOT[i]}"></div>${name}</div>`;});
  if(owner.includes(null))h+=`<div class="legend-item"><div class="ldot" style="background:#f2f2f0"></div>House</div>`;
  h+='</div>';
  const noteExtra=extra>0?` With ${vn.length} participants, ${extra} player${extra>1?'s':''} get${extra===1?'s':''} ${base+1} squares; the rest get ${base}.`:'';
  h+=`<div class="rules-box"><strong>How to find the winner:</strong> At each checkpoint, take the last digit of each team's score. Find where the home digit (column) meets the away digit (row) — that square wins. Example: a 2–1 scoreline means column 2, row 1.<br><br><strong>Digits are randomized</strong> — make sure everyone sees the final grid before the match.${noteExtra}</div>`;
  const res=document.getElementById('result');
  res.innerHTML=h;res.style.display='block';
}
function generate(){
  document.getElementById('gen-err').textContent='';
  const vn=names.filter(n=>n.trim());
  if(vn.length<1){document.getElementById('gen-err').textContent='Add at least 1 participant.';return;}
  if(vn.length>25){document.getElementById('gen-err').textContent='Maximum 25 participants.';return;}
  const sum=payouts.reduce((a,p)=>a+(parseFloat(p.pct)||0),0);
  if(Math.round(sum)!==100){document.getElementById('gen-err').textContent='Payouts must total 100%.';return;}
  const buyin=parseFloat(document.getElementById('buyin').value)||0;
  if(buyin<1){document.getElementById('gen-err').textContent='Enter a valid buy-in amount.';return;}
  const{owner,home,away}=buildGrid(vn);
  lastNames=vn;lastOwner=owner;lastHome=home;lastAway=away;
  renderResult(vn,owner,home,away);
  document.getElementById('result').scrollIntoView({behavior:'smooth',block:'start'});
}
function reroll(){
  if(!lastNames)return;
  const{owner,home,away}=buildGrid(lastNames);
  lastOwner=owner;lastHome=home;lastAway=away;
  renderResult(lastNames,owner,home,away);
}
renderNames();renderPayouts();updatePool();
</script>
